Enquiry Form Plugin 0.8.3 Dev
A simple and customizable WordPress plugin that extends WooCommerce functionality by adding an enquiry system for products. Users can select items, add them to an enquiry cart, and submit their inquiries via a form.

111function enquiry_form_create_tables() {
112 global $wpdb;
113 $charset_collate = $wpdb->get_charset_collate();
114
115 $enquiries_table = $wpdb->prefix . 'ef_enquiries';
116 $cart_items_table = $wpdb->prefix . 'ef_cart_items';
117
118 $enquiries_sql = "CREATE TABLE $enquiries_table (
119 id mediumint(9) NOT NULL AUTO_INCREMENT,
120 subject varchar(255) NOT NULL,
121 content text NOT NULL,
122 name varchar(100) NOT NULL,
123 company varchar(100) NOT NULL,
124 email varchar(100) NOT NULL,
125 phone varchar(20) NOT NULL,
126 created_at datetime DEFAULT CURRENT_TIMESTAMP,
127 processed tinyint(1) NOT NULL DEFAULT 0,
128 status varchar(20) NOT NULL DEFAULT 'Unreplied',
129 PRIMARY KEY (id)
130 ) $charset_collate;";
131
132 $cart_items_sql = "CREATE TABLE $cart_items_table (
133 id mediumint(9) NOT NULL AUTO_INCREMENT,
134 enquiry_id mediumint(9) NOT NULL,
135 product_name varchar(255) NOT NULL,
136 quantity int NOT NULL,
137 sku varchar(100) NOT NULL,
138 PRIMARY KEY (id),
139 CONSTRAINT fk_enquiry
140 FOREIGN KEY (enquiry_id)
141 REFERENCES $enquiries_table(id)
142 ON DELETE CASCADE
143 ) $charset_collate;";
144
145 require_once(ABSPATH . 'wp-admin/includes/upgrade.php');
146 dbDelta($enquiries_sql);
147 dbDelta($cart_items_sql);
148}
